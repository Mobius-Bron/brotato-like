============================================
  BrotatoLike 项目结构文档
============================================

项目根目录: brotato-like/

core/                           # 程序代码（GDScript）
  ecs/                          # ECS 框架核心
    ecs_world.gd                # ECS 世界容器（实体+组件+系统+xp_orbs）
  systems/                      # 系统（处理逻辑）
    movement_system.gd          # 移动 + 敌人互相避让 + 追踪玩家
    weapon_system.gd            # 自动瞄准+射击+近战动画(砍/刺/钝)+暴击+吸血
    weapon_orbit_system.gd      # 武器环绕玩家轨道计算
    collision_system.gd         # 碰撞检测（支持穿透）
    health_system.gd            # 生命/死亡/无敌/闪避/回复
    spawn_system.gd             # 权重制出怪 + 远离玩家随机
    render_system.gd            # 伪3D阴影 + 描边 + 形状渲染
    lifetime_system.gd          # 临时实体过期销毁
  managers/                     # 全局管理器
    game_manager.gd             # 游戏状态管理（7状态 + 20属性）
    event_bus.gd                # 全局事件系统（15种事件）
    config_loader.gd            # JSON加载器（6种配置）
    multiplayer_manager.gd      # LAN联机管理器（ENet）
  factories/                    # 工厂类
    enemy_factory.gd            # 敌人工厂 + 复合精灵
    weapon_factory.gd           # 武器属性加成（攻速/伤害%/melee_type）
    projectile_factory.gd       # 子弹工厂（支持穿透）
  ui/                           # UI 脚本
    main_menu_ui.gd             # 主菜单（单人/联机/退出）
    lobby_ui.gd                 # 联机大厅（Host/Join）
    character_select.gd         # 角色选择（13角色卡片）
    game_world.gd               # 核心游戏世界（XP球拾取+复活）
    render_layer.gd             # 渲染桥接层
    shader_rect.gd              # 泥地shader层
    shop_ui.gd                  # 商店（背包+合成+交换+刷新）
    perk_select.gd              # 波次结束Perk选择
    results_overlay.gd          # 胜利/失败结算
  shaders/                      # Shader
    mud_ground.gdshader         # 6层FBM泥地
    grass.gdshader              # (已废弃)

maps/                           # Godot 场景文件
  main_menu.tscn                # 主菜单场景
  lobby.tscn                    # 联机大厅场景
  character_select.tscn         # 角色选择场景
  game_world.tscn               # 核心战斗场景
  shop_overlay.tscn             # 商店覆盖层(左商店右背包)
  perk_select_overlay.tscn      # 波次结束Perk选择
  results_overlay.tscn          # 结算界面

data/                           # JSON 配置数据
  enemies.json                  # 21小怪 + 11 Boss
  weapons.json                  # 50+ 武器（8生态系×3 tier + melee_type）
  items.json                    # 25 道具
  waves.json                    # 20 波
  characters.json               # 13 角色(独特皮肤+差异化机制)
  perks.json                    # 26 Perk

docs/                           # 设计文档
  01-project-structure.txt
  02-data-structures.txt
  03-game-flow.txt
  04-architecture.txt
  05-run-structure.txt
